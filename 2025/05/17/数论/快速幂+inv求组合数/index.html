快速幂+乘法逆元求组合数

快速幂取long long fpow(long long a, long long b, long long mode){ long long sum = 1; while (b) { if (b & 1) { sum = (sum * a) % mode; b--; } b /= 2; a = a * a % mode; } return sum;}

乘法逆元
乘法逆元1.概念 ​ 在取模的条件下，除以一个数等价于乘以这个数的乘法逆元，比如a/b%mod=a*inv(b)%mod 2.快速幂求逆元（费马小定理） ​ p为质数的时候，b**p=b(modp) ​ 推论：模数p为质数的时候，b**(p-2)就是b的乘法逆元，直接用快速幂计算就行 ll fpm(ll a,ll b) { a%=mod; ll ans = 1; while (b) { if (b & 1) ans = (ans * a) % mod; b >>= 1; a = (a * a) % mod; } return ans;}ll inv(int x){ return fpm(x,mod-2);}
